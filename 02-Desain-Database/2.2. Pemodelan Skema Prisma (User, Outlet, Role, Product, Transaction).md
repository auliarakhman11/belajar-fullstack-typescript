# 2.2. Pemodelan Skema Prisma: Granular Multi-Outlet

Pada tahap ini, kita akan mendefinisikan "Blueprint" atau cetak biru database. Kita akan menggunakan pendekatan **Explicit Many-to-Many** untuk mengatur hak akses yang sangat spesifik (Granular Access Control) sesuai kebutuhan bisnis aplikasi POS Enterprise Anda.

---

## 1. Kode Skema Prisma (`schema.prisma`)

Salin seluruh kode di bawah ini ke dalam file `prisma/schema.prisma` di folder `pos-backend`.

```prisma
// 1. GENERATOR: Alat untuk menciptakan Prisma Client agar bisa digunakan di TypeScript
generator client {
  provider = "prisma-client-js"
}

// 2. DATASOURCE: Koneksi ke PostgreSQL (Nilai URL diambil dari file .env)
datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

// 3. ENUM ROLE: Batasan peran pengguna dalam sistem
enum Role {
  SUPER_USER // Akses penuh (Global), berwenang mengatur penugasan Admin/Kasir ke Cabang
  ADMIN      // Mengelola cabang-cabang tertentu yang ditugaskan oleh Super User
  CASHIER    // Melayani transaksi hanya di satu cabang utama yang ditentukan
}

// 4. MODEL USER: Identitas Pengguna
model User {
  id        String   @id @default(uuid()) // Menggunakan UUID agar ID unik dan tidak bisa ditebak
  name      String
  email     String   @unique
  password  String   // Akan disimpan dalam bentuk hash (misal: bcrypt)
  role      Role     @default(CASHIER)
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt

  // RELASI: Menghubungkan User ke daftar Outlet yang boleh ia akses melalui Pivot Table
  assignedOutlets UserOutlet[]

  // RELASI: Riwayat transaksi yang dilayani oleh user ini
  transactions    Transaction[]
}

// 5. MODEL OUTLET: Identitas Cabang/Toko
model Outlet {
  id        String   @id @default(uuid())
  name      String
  address   String?
  phone     String?
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt

  // RELASI: Daftar user (staff) yang bertugas atau memiliki akses ke outlet ini
  staffs    UserOutlet[]

  // RELASI: Produk dan Penjualan yang terdaftar di outlet ini
  products  Product[]
  sales     Transaction[]
}

// 6. TABEL PERANTARA (PIVOT TABLE): UserOutlet
// Inilah inti dari logika "Siapa boleh akses Cabang Mana"
model UserOutlet {
  userId    String
  user      User     @relation(fields: [userId], references: [id])

  outletId  String
  outlet    Outlet   @relation(fields: [outletId], references: [id])

  // Flag untuk menandai outlet utama (sangat krusial untuk role CASHIER)
  isPrimary Boolean  @default(false)

  // Primary Key Gabungan: Mencegah user yang sama ditugaskan dua kali di outlet yang sama
  @@id([userId, outletId])
}

// 7. MODEL PRODUCT: Katalog Barang
model Product {
  id        String   @id @default(uuid())
  name      String
  sku       String   @unique             // Barcode atau Kode barang unik
  price     Decimal  @db.Decimal(15, 2)  // Decimal menjamin presisi perhitungan uang (100% akurat)
  stock     Int      @default(0)
  imageUrl  String?                      // Menyimpan URL gambar untuk fitur Lazy Loading

  outletId  String
  outlet    Outlet   @relation(fields: [outletId], references: [id])

  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt
}

// 8. MODEL TRANSACTION: Header Penjualan Kasir
model Transaction {
  id          String   @id @default(uuid())
  invoiceNo   String   @unique           // Nomor struk unik (contoh: INV-2026-0001)
  totalAmount Decimal  @db.Decimal(15, 2)
  payment     Decimal  @db.Decimal(15, 2)
  change      Decimal  @db.Decimal(15, 2)

  userId      String                     // Kasir yang melayani saat itu
  user        User     @relation(fields: [userId], references: [id])

  outletId    String                     // Lokasi cabang tempat transaksi terjadi
  outlet      Outlet   @relation(fields: [outletId], references: [id])

  createdAt   DateTime @default(now())
}

```
